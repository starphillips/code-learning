function romanToInt(s: string): number {
    
    const romans = new Map  <string, number>([
        ['I', 1],
        ['V', 5],
        ['X', 10],
        ['L', 50],
        ['C', 100],
        ['D', 500],
        ['M', 1000],
    ]);

    let total = 0
    for (let i = 0; i < s.length; i++){
        let curr = romans.get(s[i])
        let next = romans.get(s[i+1])

        if (curr < next){
            total -= curr
        } else {
            total += curr
        }
    } 
    return total
};
