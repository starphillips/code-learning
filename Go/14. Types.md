
``` go
package main

import "fmt"

// TYPE ASSERTIONS

func main() {
	var i interface{} = "hello" // i is the interface

	s := i.(string) // string is the type youre expecting
	fmt.Println(s) // so if i is a string it will print hello

	s, ok := i.(string) // ok is a boolean 
	fmt.Println(s, ok) // so  if it doesn't work it will not crash but will say true if it worked

	f, ok := i.(float64) // false if it doesn't - here it takes float64 but the boolean makes it not error
	fmt.Println(f, ok)

	f = i.(float64) // panic
	fmt.Println(f) // no ok back up, so will just error out
}
```

**TYPE SWITCHES**

Like a switch statement but it specifies type not value 
This type is compared against the type of value held by the given interface
``` go
func do(i interface{}) {
	switch v := i.(type) {
	case int:
		fmt.Printf("Twice %v is %v\n", v, v*2) // in the case do is a int this will print
	case string:
		fmt.Printf("%q is %v bytes long\n", v, len(v))
	default:
		fmt.Printf("I don't know about type %T!\n", v) // will take any type in, T means it will recognise and print what type to go with the sentence
	}
}

func main() {
	do(21) // here it is an int so it will print the case int 
	do("hello")
	do(true) // I don't know about type bool
	do(7.5) // I don't know about type float64
}

```
