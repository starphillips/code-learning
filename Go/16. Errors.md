
``` go
package main

import (
	"fmt"
	"time"
)

type MyError struct {
	When time.Time
	What string
} // Creates a struct type

func (e *MyError) Error() string {
	return fmt.Sprintf("at %v, %s",
		e.When, e.What) // Will print this as a string using the pairs in the MyError struct
}

func run() error {
	return &MyError{ // Pointer to say the time now will go into the Error method above and will return the time now in its string
		time.Now(),
		"it didn't work", // and that it didn't work
	}
}

func main() {
	if err := run(); err != nil { // if err is the func run, and the err is not nil (there is a error) then print it
		fmt.Println(err)
	}
}
```

**ERROR EXERCISE**
``` go
package main

import (
	"fmt"
	"math"
)

type ErrNegativeSqrt float64

func (e ErrNegativeSqrt) Error() string {
	return fmt.Sprintf("can't Sqrt negative number %v", float64(e))
}


func Sqrt(x float64) (float64, error) {
	if x < 0 {
	return 0, ErrNegativeSqrt(x)
	}
	return math.Sqrt(x), nil
}

func main() {
	fmt.Println(Sqrt(2))
	fmt.Println(Sqrt(-2))
}
```
