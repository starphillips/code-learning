
```go
package main

import "fmt"

type Person struct {
	Name string
	Age  int
}

// You know its a stringer when the method is named string, has no params and returns a string
func (p Person) String() string { // Nothing but string
	return fmt.Sprintf("%v (%v years)", p.Name, p.Age) // Sprintf builds a formatted string and returns it
}

func main() {
	a := Person{"Arthur Dent", 42}
	z := Person{"Zaphod Beeblebrox", 9001}
	fmt.Println(a, z)
}

```

**EXERCISE**
``` go
package main

type IPAddr [4]byte // Here we define a type, which is a 4 byte array
// similar underlying data to --> ip := IPAddr{127, 0, 0, 1}

func (p IPAddr) String() string { // Create the stringer using the type you've made
	return fmt.Sprintf("%d.%d.%d.%d", p[0], p[1], p[2], p[3]) // Format where %d is an int and each of the array can go inside the p
}

func main() {
	hosts := map[string]IPAddr{
		"loopback":  {127, 0, 0, 1},
		"googleDNS": {8, 8, 8, 8},
	}
	for name, ip := range hosts {
		fmt.Printf("%v: %v\n", name, ip)
	}
}
```
